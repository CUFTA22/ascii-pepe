## Installation

```
npm install --save ascii-pepe
```

## Usage

```js
const pepe = require("ascii-pepe");

// As a string:
console.log(pepe.toString());

// As a colored string ( nice for browser ):
console.log(pepe.toColoredString());

// As an array:
pepe.toArray().forEach(function (line) {
  console.log(line);
});
```
